CREATE TABLE riders (id INT NOT NULL AUTO_INCREMENT, name VARCHAR(100) NOT NULL, vehicle VARCHAR(50) NOT NULL, total_deliveries INT NOT NULL DEFAULT 0, PRIMARY KEY (id)); CREATE TABLE reviews (id INT NOT NULL AUTO_INCREMENT, rider_id INT NOT NULL, customer_name VARCHAR(100) NOT NULL, rating INT NOT NULL, comment VARCHAR(500) DEFAULT NULL, PRIMARY KEY (id), FOREIGN KEY (rider_id) REFERENCES riders(id), CHECK (rating BETWEEN 1 AND 5));
